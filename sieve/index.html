/************ Eager, finite Sieve ************/

int[] primes = primesUpTo(25);

for (int prime : primes) {
    System.out.println(prime);
}

// new boolean[25]
//                  AND
// new int[9]

/*********** Lazy, infinite Sieve ***********/

Iterable<Long> primes = allPrimes();

for (long prime : primes) {
    System.out.println(prime);
}

// new       HashMap<Number, List<Prime>>
//                                         OR
// new PriorityQueue<Pair<Number, Prime>>
